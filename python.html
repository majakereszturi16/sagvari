7. Bemenet: Két tört számlálója és nevezője.
Kimenet: A két tört összege és szorzata.

# bekérem a számokat

elso_szamlalo = int( input( "Első szám számlálója: " ) );
elso_nevezo = int( input( "Első szám nevezője: " ) );

masodik_szamlalo = int( input( "Második szám számlálója: " ) );
masodik_nevezo = int( input( "Második szám nevezője: " ) );

# kiszámolom

osszeadas = str( ( ( elso_szamlalo * masodik_nevezo ) + ( elso_nevezo * masodik_szamlalo ) ) ) + "/" + str( ( elso_nevezo * masodik_nevezo ) );

szorzas = str( ( elso_szamlalo * masodik_szamlalo ) ) + "/" + str( ( elso_nevezo * masodik_nevezo ) );

# kiírom

print( "A két tört összege: " + osszeadas );

print( "A két tört szorzata: " + szorzas );
